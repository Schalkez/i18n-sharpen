    export async function parseFile(
      source: string,
      filePath: string,
      matchFunctions: string[],
      matchAttributes: string[],
      cwd: string
    ): Promise<{ result: ParsedFileResult; errors: FileParseError[] }> {

          return parseVueFile(source, filePath, matchFunctions, matchAttributes, cwd)
        case ".svelte":
          return parseSvelteFile(source, filePath, matchFunctions, matchAttributes, cwd)
        case ".astro":

        case ".ts":
        case ".tsx":
        case ".js":
        case ".jsx":

        default:
          // PARSE-06: unknown extension → empty result, NEVER throws.
          return Promise.resolve({
            result: { usedKeys: [], dynamicCalls: [], hardcodedCandidates: [] },
            errors: []
          })
      }
    }

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| file extension → parser selection | `filePath` extension chooses which parser runs; a fixed allowlist switch, no path-derived dynamic dispatch |
| unknown/unsupported file → dispatcher | Files with extensions outside the allowlist must be handled without crashing the scan |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-D1 | Denial of Service | unknown extension crashes the scan | mitigate | The `default:` branch returns an empty `ParsedFileResult` with no errors and never throws — an unsupported file cannot abort a run (asserted by the unknown-ext test). |
| T-03-D2 | Tampering | extension string drives arbitrary code path | mitigate | Routing is a fixed `switch` over a closed allowlist of literal extensions; there is no dynamic `require`/import driven by `filePath` or `source` content. Each branch calls a statically-imported parser. |
| T-03-D3 | Elevation of Privilege | dispatcher executing source | mitigate | `parseFile` only selects and delegates to parser-only functions (no `eval`, no codegen); it adds no execution surface of its own (grep asserts no `eval`). |
| T-03-D4 | Denial of Service | a single framework parser throwing aborts batch | accept | Fatal `missing-dependency` errors deliberately propagate (FW-05); file-level syntax errors are already collected by each parser (D-17). Batch-level isolation/concurrency is Phase 4's concern (out of scope here). |
</threat_model>
