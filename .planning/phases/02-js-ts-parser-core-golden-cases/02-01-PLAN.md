```typescript
export function parseTypeScriptFile(
  source: string,
  filePath: string,
  matchFunctions: string[],
  matchAttributes: string[],
  cwd: string
): { result: ParsedFileResult; errors: FileParseError[] } {
  // PERF-02: lazy-load TypeScript from the USER's workspace, never bundled.
  const ts = loadWorkspaceDep("typescript", cwd) as typeof import("typescript")

  const ext = path.extname(filePath).toLowerCase()
  const scriptKindMap: Record<string, ts.ScriptKind> = {
    ".ts": ts.ScriptKind.TS,    // JSX-DISABLED → <T> generics parse as type args (TEST-03 mechanism)
    ".tsx": ts.ScriptKind.TSX,  // JSX-ENABLED
    ".js": ts.ScriptKind.JS,
    ".jsx": ts.ScriptKind.JSX
  }
  const scriptKind = scriptKindMap[ext] ?? ts.ScriptKind.TS

  const sourceFile = ts.createSourceFile(
    filePath,
    source,
    ts.ScriptTarget.Latest,  // = 99 (ESNext) — broadest syntax support
    false,                   // setParentNodes: false — always pass sourceFile to getText/getStart
    scriptKind
  )

  const usedKeys: ParsedFileResult["usedKeys"] = []
  const dynamicCalls: ParsedFileResult["dynamicCalls"] = []
  const hardcodedCandidates: ParsedFileResult["hardcodedCandidates"] = []

  // --- helpers used by the visitor (getTagName lands here; detection helpers added in 02-02) ---
  function getTagName(tagExpr: ts.JsxTagNameExpression): string {
    if (ts.isIdentifier(tagExpr)) return tagExpr.text.toLowerCase()
    if (ts.isPropertyAccessExpression(tagExpr)) return tagExpr.getText(sourceFile).toLowerCase()
    return ""
  }

  function visit(node: ts.Node): void {
    // SKIP_TAGS: do not recurse into a skip-tag subtree (D-10).
    if (ts.isJsxElement(node)) {
      if (SKIP_TAGS.has(getTagName(node.openingElement.tagName))) return
    }
    if (ts.isJsxSelfClosingElement(node)) {
      if (SKIP_TAGS.has(getTagName(node.tagName))) return
    }

    // NOTE (02-02): call / attribute / jsxText / jsxExpression detection is added here.

    ts.forEachChild(node, visit)
  }

  ts.forEachChild(sourceFile, visit)
  return { result: { usedKeys, dynamicCalls, hardcodedCandidates }, errors: [] }
}
```

- `errors` is `[]` this phase (research Pattern 10, approach B — the parser never throws; `ts.createSourceFile` is always-recovering). Do not read internal `parseDiagnostics`.

Same for `HARDCODED_ATTRS` / `usedKeys` / `dynamicCalls` / `hardcodedCandidates` if unused-warnings block the build — but they ARE referenced (returned), so only the two params need the `void` guard.
