<objective>
Add a `sortKeys` config option + `--sort` CLI flag that controls key ordering when `extract` and `prune` write locale files. Three modes:
- `"alpha"`: case-insensitive, numeric-aware Unicode sort via `Intl.Collator('en', { sensitivity: 'base', numeric: true })` (D-03)
- `"source"`: insertion order from `detectUsedKeys` Set (D-05) — requires `getFiles` to sort directory entries (RG-2)
- `"preserve"` (default): identity, zero behavior change vs v0.2.x (D-03 default, SORT-03)

Sorts recurse into nested objects (D-04) and apply in both `flat` and `namespaced` layouts, in both `extract` and `prune` (D-06). The CLI flag overrides the config field per invocation (D-02). Includes property-based tests proving sort is idempotent and never loses or adds keys (D-11).

Purpose: Deterministic key ordering across cross-machine CI runs and across locale languages, enabling reviewable diffs and stable git history.
Output: New `sort.ts` module + integration into extract/prune write paths + comprehensive sort tests.
</objective>

    ```typescript
    /**
     * Locale key ordering mode.
     * - "alpha": case-insensitive, numeric-aware Unicode sort via Intl.Collator('en', { sensitivity: 'base', numeric: true })
     * - "source": order keys by their position in the provided `keyOrder` Set; keys not in the set go to the end in insertion order
     * - "preserve": identity — return the input object reference unchanged (zero diff vs v0.2.x)
     */
    export type SortMode = "alpha" | "source" | "preserve"

    const ALPHA_COLLATOR = new Intl.Collator("en", { sensitivity: "base", numeric: true })

    /**
     * Recursively sort keys in a locale object.
     *
     * - "preserve" returns the input by reference (no allocation).
     * - "alpha" uses the module-level `ALPHA_COLLATOR` (fixed 'en' locale for cross-machine determinism per D-03).
     * - "source" orders keys by their position in `keyOrder` (a Set whose iteration order = first-seen order from `detectUsedKeys`).
     *   Keys NOT in `keyOrder` are appended at the end in their original insertion order.
     *
     * For nested objects, this function recurses into each `Record<string, unknown>` value with the SAME mode and the SAME keyOrder.
     * Non-object values (strings, numbers, booleans, null, arrays) are returned by reference unchanged.
     *
     * Note for source mode: `keyOrder` typically contains DOTTED-FLAT keys (e.g. "auth.login.title") but nested-object keys are SEGMENTS (e.g. "login"). At each recursion level, compare the current key against the prefix segments of `keyOrder` entries — i.e., if any keyOrder entry starts with `<currentPath>.<key>` or equals `<currentPath>.<key>`, treat the key as ordered. To keep the implementation simple and correct, the executor MAY collect the FIRST-APPEARANCE index of each segment at each recursion level by scanning keyOrder once and building a `Map<string, number>` keyed by the segment-at-current-depth. The exact strategy is the executor's choice provided the behavior matches the test cases.
     */
    export function sortLocaleObject(
      obj: Record<string, unknown>,
      mode: SortMode,
      keyOrder?: Set<string>
    ): Record<string, unknown> {
      if (mode === "preserve") return obj

      const keys = Object.keys(obj)
      let orderedKeys: string[]

      if (mode === "alpha") {
        orderedKeys = [...keys].sort((a, b) => ALPHA_COLLATOR.compare(a, b))
      } else {
        // source mode
        const order = keyOrder ?? new Set<string>()
        const indexOf = new Map<string, number>()
        let i = 0
        for (const entry of order) {
          // For top-level: the segment is the part before the first '.'
          // The recursive caller is responsible for passing a scoped keyOrder; for the top-level call the executor builds the segment index by extracting the first segment of each entry.
          const seg = entry.includes(".") ? entry.slice(0, entry.indexOf(".")) : entry
          if (!indexOf.has(seg)) indexOf.set(seg, i++)
        }
        orderedKeys = [...keys].sort((a, b) => {
          const ia = indexOf.get(a)
          const ib = indexOf.get(b)
          if (ia === undefined && ib === undefined) return 0  // both unknown — preserve insertion order
          if (ia === undefined) return 1                       // unknown goes after known
          if (ib === undefined) return -1
          return ia - ib
        })
      }

      const out: Record<string, unknown> = {}
      for (const k of orderedKeys) {
        const v = obj[k]
        if (v !== null && typeof v === "object" && !Array.isArray(v)) {
          out[k] = sortLocaleObject(v as Record<string, unknown>, mode, keyOrder)
        } else {
          out[k] = v
        }
      }
      return out
    }
    ```
